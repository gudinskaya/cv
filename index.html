Primality test

          
  const isPrime = (n) => {
    if (n < 2) {
      return false
    }
  
    let i = 2
  
    while (i * i <= n) {
      if (n % i === 0) {
        return false
      }
      i += 1
    }
    return true
  }

Algorithm that finds greatest common divisor

          
  const mygcd = (x, y) => {
    let x1 = x
    let y1 = y
    while (y1 != 0) {
      const tmp = x1
      x1 = y1
      y1 = tmp % y1
    }
    return x1
  }
